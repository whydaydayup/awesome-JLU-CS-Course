{

void Swap(int &m,int &n)
{
 int temp=m;
 m=n;
 n=temp;
}
void Qsort(int *R,int m,int n)
{
if(m<n)
{
	int i=m;
	int j=n+1;
	while(i<j)
	{
      while(R[i]<R[m]) i++;
	  j--;
	  while(R[j]>R[m]) j--;
	  if(i<j) Swap(R[i],R[j]);
	}
	Swap(R[m],R[j]);

	Qsort(R,m,j-1);
	Qsort(R,j+1,n);
}
}
